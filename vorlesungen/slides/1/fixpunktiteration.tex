\begin{frame}
\frametitle{Fixpunktiteration}

\begin{block}{Folge}
Gegeben: Stetige Funktion $f\colon \mathbb R\to\mathbb R:x\mapsto f(x)$,
$x_0\in \mathbb R$

Iterationsfolge:
\[
x_{n+1} = f(x_n)
\]
\end{block}
\uncover<2->{
\begin{block}{Möglichkeiten}
\begin{enumerate}
\item<3-> Konvergenz: $\displaystyle \lim_{n\to\infty} x_n=x^*$ ist Fixpunkt: $f(x^*)=x^*$, Schnittpunkt $\operatorname{Graph}(f)$ mit $y=x$
\item<4-> Divergenz: $\displaystyle \lim_{n\to\infty} x_n=\pm\infty$
\item<5-> Chaos
\item<6-> Periodische Punkte: $x_n$ konvergiert gegen eine Zyklus der Länge
$k$ für $k\in\mathbb N$.
\end{enumerate}
\end{block}}

\end{frame}
