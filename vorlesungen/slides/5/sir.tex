\begin{block}{Differentialgleichung}
\[
\left.
\begin{aligned}
\frac{dS}{dt} &=          - \beta \, SI      \\
\frac{dI}{dt} &= \phantom{-}\beta \, SI - \gamma I \\
\frac{dR}{dt} &= \phantom{- \beta \, SI}+ \gamma I 
\end{aligned}
\quad
\uncover<3->{%
\right\}
\quad
\Rightarrow
\quad
\frac{d}{dt} (S+I+R) = 0
\Rightarrow
S+I+R=\text{const}}
\]
