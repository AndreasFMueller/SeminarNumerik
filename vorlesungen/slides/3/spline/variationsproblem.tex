\frametitle{Variationsprobleme}
\begin{columns}[t]
\begin{column}{0.48\hsize}
\begin{block}{Aufgabe}
Gegeben die Funktion $L(t,x,v)$, finde eine Kurve $x(t)$ mit 
$x(a) = x_a$ und $x(b)=x_b$ derart, dass die Wirkung
\[
W
=
W(x)
=
\int_a^b L(t,x(t), \dot x(t))\,dt
\]
minimal ist.
\end{block}
\end{column}
