\frametitle{Integral $\rightarrow$ Gleichung}
\vspace{-15pt}
\begin{columns}[t]
\begin{column}{0.48\hsize}
\begin{block}{Prinzip}
$f(x)$ eine stetige Funktion in $[a,b]$ und
\[
\int_a^b f(x) h(x) \,dx = 0\quad \forall h
\]
dann ist $f(x)=0$
\end{block}
\uncover<2->{%
\begin{proof}[Beweis]
\begin{enumerate}
\item<3->
Annahme: $f(\bar{x}) \only<-6>{>}\only<7->{<} 0$
\item<4->
$f(x) \only<-6>{>}\only<7->{<} 0$ f체r $|x-\bar{x}|\le \varepsilon$
\item<5->
W채hle $h(x)$ mit $h(x) > 0$ f체r $|x-\bar{x}|<\varepsilon$, $f(x)=0$ sonst.
\item<6->
$\int_a^b f(x)h(x) \,dx \only<-6>{>}\only<7->{<} 0\; \Rightarrow $ Widerspruch!
\qedhere
\end{enumerate}
\end{proof}}
\end{column}
