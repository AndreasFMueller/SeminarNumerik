\vspace{-20pt}
\begin{columns}[t]
\begin{column}{0.51\hsize}
\begin{block}{Aufgabe}
Für $f\colon[a,b]\to\mathbb R$ stetig berechne
\vspace{-5pt}
\[
I = \int_a^b f(x)\,dx
\]
\end{block}
\vspace{-15pt}
\uncover<2->{%
\begin{block}{Interpretationen}
\begin{itemize}
\item<3-> Fläche unter der Kurve
\item<6-> Stammfunktion $F(x)$ mit $F'(x)=f(x)$
$\Rightarrow$
$I = F(b)-F(a)$
\item<7->
Lösung $y(x)$ der Differentialgleichung%
\vspace{-5pt}
\begin{align*}
y' &= f(x),&y(a)&=0\\[-20pt]
\end{align*}
$\Rightarrow$ $I = y(b)$
\end{itemize}
\end{block}}
\end{column}
