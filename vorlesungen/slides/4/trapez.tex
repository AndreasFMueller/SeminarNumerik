\uncover<2->{%
\begin{block}{Unterteilung}
Gleichlange Teilintervalle:
\[
x_k=a+k\frac{b-a}{n}=a+kh
\]
\end{block}}
\vspace{-10pt}
\uncover<4->{%
\begin{block}{Trapeze statt Rechtecke}
\vspace{-10pt}
\[
f(\tilde{x}_k) h \rightarrow \frac{f(x_{k-1})+f(x_k)}{2}h
\]
\end{block}}

		--

		--cycle;
