\begin{column}{0.40\hsize}
\uncover<2->{%
\begin{block}{Unterteilung}
\vspace{-20pt}
\[
a=x_0 < x_1 < \dots < x_{n-1} < x_n = b
\]
\end{block}}
\vspace{-10pt}
\uncover<3->{%
\begin{block}{Maximal- und Minimalwerte}
\vspace{-20pt}
\begin{align*}
\overline{f}_k
&=\max\{
f(x)\;|\;
x\in[x_{k-1},x_k]
\}
\\
\underline{f}_k
&=
\min\{
f(x)\;|\;
x\in[x_{k-1},x_k]
\}
\end{align*}
\end{block}}
\vspace{-10pt}
\uncover<4->{%
\begin{block}{Ober- und Untersummen}
\vspace{-15pt}
\begin{align*}
\underline{I}
&=
\sum_{k=1}^n \underline{f}_k (x_k-x_{k-1})
\\
\overline{I}
&=
\sum_{k=1}^n \overline{f}_k (x_k-x_{k-1})
\end{align*}
\end{block}}
\end{column}
