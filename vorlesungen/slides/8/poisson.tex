\uncover<3->{%
\begin{block}{Approximation der Gleichung $\Delta u=f$}
\vspace{-15pt}
\begin{align*}
-4
u_{ij}
+
u_{i-1,j}+u_{i+1,j}+u_{i,j-1}+u_{i,j+1}
&=
h^2f_{ij}
\\
\frac{1}{h^2} Lu &= f
\end{align*}
%%$\rightarrow$ {\em 5-Punkte-Stern}
\end{block}}
\vspace{-10pt}

\vspace{-10pt}

u_{ij} = {\color{blue}g_{ij}}
\quad
\text{für Randpunkte $(x_i,y_j)$}
\]
\end{block}
}
