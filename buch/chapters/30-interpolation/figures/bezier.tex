\begin{scope}[xshift=0cm]

\def\qunkt#1{({(1-#1)*1+#1*4},{(1-#1)*5+#1*1})}
\def\runkt#1{({(1-#1)*(1-#1)*0+2*#1*(1-#1)*1+#1*#1*4},{(1-#1)*(1-#1)*0+2*#1*(1-#1)*5+#1*#1*1})}

	plot[domain=0:1,samples=100]
		({(1-\x)*(1-\x)*0+2*\x*(1-\x)*1+\x*\x*4},{(1-\x)*(1-\x)*0+2*\x*(1-\x)*5+\x*\x*1});
